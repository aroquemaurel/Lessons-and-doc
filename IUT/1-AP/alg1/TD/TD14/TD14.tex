		\section{}	
		\section{Postcondition}	
			$$ \forall i \in [1,n-1], tab[i] < tab[i+1]$$ 
		\section{Comparaisons}
		\begin{eqnarray*}
			&&(n-1)+(1-2)+(1-3)+\dots+1\\
			&=&\sum^{n-1}_{c=1}{(n-1)} \\
			&=&\sum^{n-1}_{i=n}-\sum^{n-1}_{i=1}{i} \\
			&=& \frac{(n-1)(n+2)}{2} - (n-1) = 
			\frac{n(n-1)}{2}=O(n^{2})
		\end{eqnarray*} 
			Donc complexité en $O(n^{2})$
